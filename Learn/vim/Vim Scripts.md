
```
" Disable compatibility with vi
set nocompatible

" Enable file type detection, plugins, and indenting
filetype on
filetype plugin on
filetype indent on

" Enable syntax highlighting
syntax enable

" Set the colorscheme (e.g., 'molokai', 'solarized', 'desert')
colorscheme one " or your preferred colorscheme
set termguicolors

" Show line numbers
set number

set cursorline
set cursorcolumn

" Enable mouse support
set mouse=a

" Set tab and indent settings
set tabstop=4
set shiftwidth=4
set expandtab

set nobackup

set ignorecase
set smartcase

set incsearch
set hlsearch

inoremap jj <Esc>

```
